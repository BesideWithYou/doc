## 实现一个栈

```rust
#[derive(Debug)]
struct Stack<T> {
    size: usize,
    inner: Vec<T>,
}

impl<T> Stack<T> {
    fn new() -> Self {
        Stack {
            size: 0,
            inner: vec![],
        }
    }

    fn push(&mut self, value: T) {
        self.inner.push(value);
        self.size += 1;
    }

    fn pop(&mut self) -> Option<T> {
        if self.size == 0 {
            return None;
        }

        self.size -= 1;
        self.inner.pop()
    }

    fn peek(&self) -> Option<&T> {
        if self.size == 0 {
            return None;
        }

        self.inner.get(self.size - 1)
    }

    fn is_empty(&self) -> bool {
        self.size == 0
    }

    fn size(&self) -> usize {
        self.size
    }
}

fn main() {
    let mut stack = Stack::new();
    stack.push(1);
    stack.push(2);
    // 打印 2
    println!("{:#?}", stack.peek().unwrap());
    stack.pop();
    // 打印 1
    println!("{:#?}", stack.peek().unwrap());
}
```
