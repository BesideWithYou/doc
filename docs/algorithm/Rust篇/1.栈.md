# 实现一个栈

```rust
/// 栈
#[derive(Debug)]
struct Stack<T> {
    size: usize,
    data: Vec<T>,
}

/// 实现栈
impl<T> Stack<T> {
    fn new() -> Self {
        Self {
            // 当前长度
            size: 0,
            // 数据容器
            data: vec![],
        }
    }

    /// 入栈
    fn push(&mut self, value: T) {
        self.data.push(value);
        self.size += 1;
    }

    /// 出栈
    fn pop(&mut self) -> Option<T> {
        if self.size == 0 {
            return None;
        }

        self.size -= 1;
        self.data.pop()
    }

    /// 返回栈顶的引用，用来查看数据
    fn peek(&self) -> Option<&T> {
        if self.size == 0 {
            return None;
        }

        self.data.get(self.size - 1)
    }

    /// 栈是否为空
    fn is_empty(&self) -> bool {
        self.size == 0
    }

    /// 栈的大小
    fn size(&self) -> usize {
        self.size
    }
}

fn main() {
    let mut stack = Stack::new();
    // 入栈
    stack.push(1);
    stack.push(2);

    // 打印 2，返回引用，不出栈
    println!("{:#?}", stack.peek().unwrap());

    // 出栈，返回出栈后的值
    stack.pop();
    // 打印 1
    println!("{:#?}", stack.peek().unwrap());
}
```
